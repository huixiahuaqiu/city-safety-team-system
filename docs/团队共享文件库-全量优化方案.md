@Service
@Slf4j
public class FileStorageService {

    @Value("${file.upload-path}")
    private String uploadRootPath;

    /**
     * 保存上传的文件，返回存储路径
     */
    public String saveFile(MultipartFile file, String relativeDir) throws IOException {
        // 1. 生成存储路径：按年月分目录，避免单目录文件过多
        String datePath = DateUtil.format(new Date(), "yyyyMM");
        String dirPath = uploadRootPath + relativeDir + "/" + datePath + "/";
        File dir = new File(dirPath);
        if (!dir.exists()) dir.mkdirs();

        // 2. 生成唯一文件名，防止重名覆盖
        String originalName = file.getOriginalFilename();
        String suffix = FileUtil.extName(originalName);
        String newFileName = IdUtil.simpleUUID() + "." + suffix;
        String fullPath = dirPath + newFileName;

        // 3. 写入文件
        file.transferTo(new File(fullPath));
        return relativeDir + "/" + datePath + "/" + newFileName;
    }

    /**
     * 获取文件绝对路径
     */
    public String getAbsolutePath(String relativePath) {
        return uploadRootPath + relativePath;
    }

    /**
     * 删除物理文件
     */
    public boolean deleteFile(String relativePath) {
        String fullPath = getAbsolutePath(relativePath);
        return FileUtil.del(fullPath);
    }
}

3.3 业务服务层 TeamSharedFileService.java
java
运行
@Service
@Slf4j
public class TeamSharedFileService {

    @Autowired
    private TeamSharedFileMapper fileMapper;
    @Autowired
    private FileStorageService storageService;
    @Autowired
    private FileDownloadLogMapper downloadLogMapper;

    /**
     * 分页查询文件列表
     */
    public PageResult<TeamSharedFile> getFilePage(FileQuery query) {
        Page<TeamSharedFile> page = fileMapper.selectPage(
                query.buildPage(),
                Wrappers.<TeamSharedFile>lambdaQuery()
                        .eq(StrUtil.isNotBlank(query.getFileType()), TeamSharedFile::getFileType, query.getFileType())
                        .like(StrUtil.isNotBlank(query.getKeyword()), TeamSharedFile::getFileName, query.getKeyword())
                        .eq(TeamSharedFile::getIsDeleted, 0)
                        .eq(TeamSharedFile::getStatus, 1)
                        .orderByDesc(TeamSharedFile::getCreateTime)
        );
        return PageResult.of(page);
    }

    /**
     * 上传文件
     */
    @Transactional(rollbackFor = Exception.class)
    @Log(module = "资源中心", type = "上传", desc = "上传共享文件")
    public Long uploadFile(MultipartFile file, String fileName, String fileType, String remark) throws Exception {
        // 1. 读取系统配置，校验文件大小
        int maxSizeMb = ConfigUtils.getInt("file.maxSize", 10);
        if (file.getSize() > maxSizeMb * 1024 * 1024) {
            throw new RuntimeException("文件大小不能超过" + maxSizeMb + "MB");
        }

        // 2. 校验文件类型
        String allowTypes = ConfigUtils.getString("file.allowTypes", "jpg,png,pdf,doc,docx,xls,xlsx,zip");
        String suffix = FileUtil.extName(file.getOriginalFilename()).toLowerCase();
        if (!Arrays.asList(allowTypes.split(",")).contains(suffix)) {
            throw new RuntimeException("不支持的文件类型，允许：" + allowTypes);
        }

        // 3. 保存文件到磁盘
        String storagePath = storageService.saveFile(file, "shared");

        // 4. 保存数据库记录
        TeamSharedFile record = new TeamSharedFile();
        record.setFileName(StrUtil.isNotBlank(fileName) ? fileName : file.getOriginalFilename());
        record.setOriginalName(file.getOriginalFilename());
        record.setFileType(fileType);
        record.setFileSize(file.getSize());
        record.setFileSuffix(suffix);
        record.setStoragePath(storagePath);
        record.setUploaderId(SecurityUtils.getCurrentUserId());
        record.setUploaderName(SecurityUtils.getCurrentUserName());
        record.setRemark(remark);
        fileMapper.insert(record);

        return record.getId();
    }

    /**
     * 下载文件
     */
    @Transactional(rollbackFor = Exception.class)
    public File downloadFile(Long id) {
        TeamSharedFile file = fileMapper.selectById(id);
        if (file == null || file.getIsDeleted() == 1) {
            throw new RuntimeException("文件不存在");
        }

        // 1. 权限校验：访客不能下载
        if (SecurityUtils.hasRole("visitor")) {
            throw new RuntimeException("访客账号无下载权限");
        }

        // 2. 下载次数+1
        file.setDownloadCount(file.getDownloadCount() + 1);
        fileMapper.updateById(file);

        // 3. 记录下载日志
        FileDownloadLog log = new FileDownloadLog();
        log.setFileId(id);
        log.setFileName(file.getFileName());
        log.setDownloaderId(SecurityUtils.getCurrentUserId());
        log.setDownloaderName(SecurityUtils.getCurrentUserName());
        log.setDownloadIp(IpUtils.getIpAddr(ServletUtils.getRequest()));
        downloadLogMapper.insert(log);

        // 4. 返回文件
        String fullPath = storageService.getAbsolutePath(file.getStoragePath());
        File targetFile = new File(fullPath);
        if (!targetFile.exists()) {
            throw new RuntimeException("文件已丢失");
        }
        return targetFile;
    }

    /**
     * 删除文件
     */
    @Transactional(rollbackFor = Exception.class)
    @Log(module = "资源中心", type = "删除", desc = "删除共享文件")
    public void deleteFile(Long id) {
        TeamSharedFile file = fileMapper.selectById(id);
        if (file == null) return;

        // 权限校验：管理员可以删所有；普通用户只能删自己上传的
        boolean isAdmin = SecurityUtils.hasRole("admin");
        boolean isOwner = file.getUploaderId().equals(SecurityUtils.getCurrentUserId());
        if (!isAdmin && !isOwner) {
            throw new RuntimeException("无权限删除该文件");
        }

        // 1. 删除物理文件
        storageService.deleteFile(file.getStoragePath());
        // 2. 软删除数据库记录
        file.setIsDeleted(1);
        fileMapper.updateById(file);
    }
}

@RestController
@RequestMapping("/resource/shared-file")
@Tag(name = "团队共享文件库")
public class TeamSharedFileController {

    @Autowired
    private TeamSharedFileService fileService;

    @GetMapping("/list")
    @Operation(summary = "分页查询文件列表")
    public Result<PageResult<TeamSharedFile>> list(FileQuery query) {
        return Result.success(fileService.getFilePage(query));
    }

    @PostMapping("/upload")
    @Operation(summary = "上传文件")
    public Result<Long> upload(
            @RequestParam("file") MultipartFile file,
            @RequestParam(required = false) String fileName,
            @RequestParam String fileType,
            @RequestParam(required = false) String remark
    ) throws Exception {
        Long id = fileService.uploadFile(file, fileName, fileType, remark);
        return Result.success(id);
    }

    @GetMapping("/download/{id}")
    @Operation(summary = "下载文件")
    public void download(@PathVariable Long id, HttpServletResponse response) throws IOException {
        File file = fileService.downloadFile(id);
        response.setContentType("application/octet-stream");
        response.setHeader("Content-Disposition",
                "attachment;filename=" + URLEncoder.encode(file.getName(), StandardCharsets.UTF_8));
        FileUtil.writeToStream(file, response.getOutputStream());
    }

    @DeleteMapping("/{id}")
    @Operation(summary = "删除文件")
    public Result<Void> delete(@PathVariable Long id) {
        fileService.deleteFile(id);
        return Result.success();
    }
}

// 格式化文件大小
const formatSize = (bytes) => {
  if (!bytes) return '0 B'
  const units = ['B', 'KB', 'MB', 'GB']
  let index = 0
  let size = bytes
  while (size >= 1024 && index < units.length - 1) {
    size /= 1024
    index++
  }
  return size.toFixed(2) + ' ' + units[index]
}

// 判断是否可删除
const canDelete = (item) => {
  // 管理员可以删所有；普通用户只能删自己上传的
  if (userStore.roles.includes('admin')) return true
  return item.uploaderId === userStore.userId
}

// 获取列表

  loading.value = true
  query.fileType = activeType.value
  query.keyword = searchKeyword.value
  try {
    const res = await getFileList(query)
    tableData.value = res.data.records
    total.value = res.data.total
  } finally {
    loading.value = false
  }
}

  query.pageNum = 1

}

  uploadForm.file = file.raw
  uploadForm.fileName = file.name
  uploadForm.fileSize = file.size
}

// 确认上传
const confirmUpload = async () => {
  if (!uploadForm.file) {
    ElMessage.warning('请选择要上传的文件')
    return
  }
  if (!uploadForm.fileName) {
    ElMessage.warning('请输入文件名')
    return
  }

  uploading.value = true
  try {
    const formData = new FormData()
    formData.append('file', uploadForm.file)
    formData.append('fileName', uploadForm.fileName)
    formData.append('fileType', uploadForm.fileType)
    formData.append('remark', uploadForm.remark)

    await uploadFile(formData, (percent) => {
      uploadPercent.value = percent
    })

    ElMessage.success('上传成功')
    uploadDialogVisible.value = false
    fetchList()
  } finally {
    uploading.value = false
  }
}

// 下载
const handleDownload = async (id) => {
  const res = await downloadFile(id)
  const blob = new Blob([res])
  const url = URL.createObjectURL(blob)
  const a = document.createElement('a')
  a.href = url
  a.click()
  URL.revokeObjectURL(url)
}

    type: 'warning'
  }).then(async () => {
    await deleteFile(id)
    ElMessage.success('删除成功')
    fetchList()
  })
}
